nda_ares.2 = { # The Crystal Egg
	type = country_event
	placement = ROOT
	title = nda_ares.2.t
	desc = nda_ares.2.d
	flavor =  nda_ares.2.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_london_center.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/london_center"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		is_player = yes
		has_global_variable = nda_ares_is_loaded
		#has_global_variable = nda_ares_invasion_choice
		OR = {
			AND = {
				# year before early invasion
				has_game_rule = early_landing
				#has_global_variable = nda_ares_early_invasion
				game_date >= 1841.3.1 
				game_date < 1842.1.1			
			}
			AND = {
				# year before HG Wells invasion
				has_game_rule = hg_wells_landing
				#has_global_variable = nda_ares_wells_hg_invasion
				game_date >= 1897.3.1 
				game_date < 1898.1.1			
			}
			AND = {
				# year before Orson Wells invasion
				has_game_rule = orson_wells_landing
				#has_global_variable = nda_ares_wells_orson_invasion
				game_date >= 1935.3.1 
				game_date < 1936.1.1			
			}
			AND = {
				# before immediate invasion
				has_game_rule = immediate_landing
				#has_global_variable = nda_ares_immediate_invasion				
				game_date >= 1836.2.1 
				game_date < 1837.1.1			
			}
		}
	}

	cooldown = { months = 12 }

	immediate = {
		
		debug_log = "Ares - Early Warning"
	}	
	
	option = {
		name = nda_ares.2.a
		default_option = yes
	}

}

nda_ares.6 = { # Invasion watcher event - use this to launch main event as popup
	type = country_event
	hidden = yes

	trigger = {
		OR = {
			AND = {
				# 1836 and owner of East Anglia 
				game_date >= 1836.6.1 
				game_date < 1837.1.1
				has_game_rule = immediate_landing
				p:x60A040 = { state = { owner = ROOT } }
				NOT = { has_global_variable = nda_ares_invasion }
				#has_global_variable = nda_ares_immediate_invasion
			}
			AND = {
				# 1842 and owner of East Anglia 
				game_date >= 1842.3.1 
				game_date < 1843.1.1
				p:x60A040 = { state = { owner = ROOT } }
				NOT = { has_global_variable = nda_ares_invasion }
				has_game_rule = early_landing				
			}
			AND = {
				# 1901 and owner of East Anglia 
				game_date >= 1901.3.1 
				game_date < 1902.1.1
				p:x60A040 = { state = { owner = ROOT } }
				NOT = { has_global_variable = nda_ares_invasion }
				has_game_rule = hg_wells_landing			
			}
			AND = {
				# 1936 (post standard end!) and owner of Upper New York 
				game_date >= 1936.3.1 
				game_date < 1937.1.1
				p:xB00080 = { state = { owner = ROOT } }
				NOT = { has_global_variable = nda_ares_invasion }
				has_game_rule = orson_wells_landing		
			}
		}

	}

	cooldown = { months = 24 } # should only ever run once per invasion - game rules can be adjusted during game?

	immediate = {
		debug_log = "Invasion Launch: [THIS.GetCountry.GetNameNoFormatting]" 

		if = {
			limit = { has_game_rule = immediate_landing }
			trigger_event = { id = nda_ares.13 popup = yes }			
		}
		else_if = {
			limit = { has_game_rule = early_landing }
			trigger_event = { id = nda_ares.10 popup = yes }			
		}
		else_if = {
			limit = { has_game_rule = hg_wells_landing }
			trigger_event = { id = nda_ares.11 popup = yes }			
		}
		else_if = {
			limit = { has_game_rule = orson_wells_landing }
			trigger_event = { id = nda_ares.12 popup = yes }			
		}
	}

}

nda_ares.10 = { # Early Landing
	type = country_event
	placement = p:x60A040 #East Anglia
	title = nda_ares.10.t
	desc = nda_ares.10.d
	flavor =  nda_ares.10.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"	
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	# trigger = {
 	# 	# 1842 and owner of East Anglia 
	# 	game_date >= 1842.3.1 
	# 	game_date < 1843.1.1
	# 	p:x60A040 = { state = { owner = ROOT } }
	# 	NOT = { has_global_variable = nda_ares_invasion }
	# 	has_game_rule = early_landing
	# }

	cooldown = { months = 12 }

	immediate = {
		
		set_global_variable = nda_ares_invasion

		# record target country ready to start war later
		p:x60A040.state.owner = {
			save_scope_as = nda_ares_first_victim
		}

		add_modifier = {
			name = nda_ares_country_martian_terror_modifier
		}

		# create country
		create_country = {
			tag = NDA_ARES
			origin = this

			state = p:x60A040.state 
			province = p:x60E0C0
			province = p:x19CECD
			province = p:x58D3AE
			province = p:x800020
			province = p:xA0A0C0
			province = p:x60E080
			province = p:x2482B4
			province = p:x528DEE

			on_created = {
				nda_effect_martian_setup = yes

				# lay claim to neighbouring states
				every_neighbouring_state = {
					limit = { NOT = { owner = c:NDA_ARES } }
					state_region = {
						add_homeland = cu:nda_cul_aresian		
						add_claim = c:NDA_ARES
					}
					add_radicals_in_state = { value = 0.6 } # Panic in the streets of London...

					add_modifier = {
						name = nda_ares_state_martian_terror_modifier
					}
				}	
				
				# Force Migrations? - via delayed events?
				trigger_event = { id = nda_ares.20 days=7 }

				IF = {
					limit = {
						exists = c:FRA
					}
					set_secret_goal = {
						country = c:FRA
						secret_goal = conquer
					}			
				}
				IF = {
					limit = {
						exists = c:USA
					}
					set_secret_goal = {
						country = c:USA
						secret_goal = conquer
					}			
				}
				IF = {
					limit = {
						exists = c:PRU
					}
					set_secret_goal = {
						country = c:PRU
						secret_goal = conquer
					}			
				}
				IF = {
					limit = {
						exists = c:RUS
					}
					set_secret_goal = {
						country = c:RUS
						secret_goal = conquer
					}			
				}
			}
			
		}

		# should not be liked by any one
		every_country = {
			limit = {
				NOT = { c:NDA_ARES ?= THIS }
			}
			change_relations = { country = c:NDA_ARES value = -150 } # is this too much ?

			# notify the world

			# panic stations
			add_radicals = { 
				value = 0.3 
			}
			add_loyalists = {
				value = -0.2
			}

			# Diplo Catalyst - try and create anti-invasion lobbies
			create_diplomatic_catalyst = {  
				type = catalyst_event_negative
				target = c:NDA_ARES
			}
			
		}

		#go to war
		c:NDA_ARES = {
			create_diplomatic_play = {
				name = nda_ares_dp_martian_invasion
			
				target_country = scope:nda_ares_first_victim

				war = yes 
				
				type = dp_annex_war # dp_conquer_state - can't be done ??

				# add_war_goal = {
				# 	holder = c:NDA_ARES
				# 	type = conquer_state
				# 	target_state = p:x7543C8.state #  London
				# }
			}
			
			# every_scope_war = {
			# 	limit = { 
			# 		is_war_participant =  scope:nda_ares_first_victim 
			# 	} 
			# 	add_war_exhaustion = { target = scope:nda_ares_first_victim value = 0.25 } 
			# 	debug_log = "Ares Victim - War Exhaustion"
			# }
			
		}

		# targets allies should have increased radicalism to try and knock them out of war
		scope:nda_ares_first_victim = {
			every_subject_or_below = {
				# panic stations
				add_radicals = { 
					value = 0.2 
				}
				add_loyalists = {
					value = -0.2
				}				
			}
			# every_scope_cobelligerent = {
			# 	# panic stations
			# 	add_radicals = { 
			# 		value = 0.3 
			# 	}
			# 	add_loyalists = {
			# 		value = -0.2
			# 	}
			# }
		}

		debug_log = "Ares - Start Completed"
	}	
	
	option = {
		name = nda_ares.10.a
		default_option = yes
	}

}
