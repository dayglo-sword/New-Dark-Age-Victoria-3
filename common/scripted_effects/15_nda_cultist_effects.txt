nda_effect_ensure_cult_ideology = { # character should have an appropriate ideology
	if = {
		limit = { has_non_cultist_ideology = yes }
		random_list = {	
			35 = { set_ideology = ideology:ideology_nda_cultist }
			20 = { set_ideology = ideology:ideology_jingoist_leader }
			15 = { set_ideology = ideology:ideology_fascist }
			20 = { set_ideology = ideology:ideology_slaver }
			#1.3
			10 = { set_ideology = ideology:ideology_authoritarian }
			10 = { set_ideology = ideology:ideology_luddite }
			#1.9
			10 = { set_ideology = ideology:ideology_despotic_utopian }			
		}
	}
}

# IG level effect
nda_effect_failing_cult_ideology = { # character should have an appropriate ideology
		# remove 'secret masters' as not appropriate
		if = {
			limit = { has_modifier = nda_ig_secret_masters_modifier }
			# remove any secret masters modifier
			remove_modifier = nda_ig_secret_masters_modifier				
		}
		if = {
			limit = { has_modifier = nda_ig_cult_justified_modifier }
			# remove any 'cult justified' modifier
			remove_modifier = nda_ig_cult_justified_modifier				
		}	
		# replace with 'cult crisis'
		add_modifier = {
			name = nda_ig_cult_crisis_modifier
		}
}

nda_effect_cult_perks = {
	# perks of the cult 
	if = {
		limit = {
			OR = {
				has_trait = sickly
				has_trait = cancer
				has_trait = syphilis
			}
		}
		remove_trait = sickly
		remove_trait = cancer
		remove_trait = syphilis	
		set_variable = nda_cured_by_cult				
	}	
}

nda_effect_add_negative_trait = {
	if = {
		limit = { has_negative_trait = no }
		random_list = {
			50 = { add_trait = psychological_affliction }
			20 = { add_trait = alcoholic }
			10 = { add_trait = cocaine_addiction }
			10 = { add_trait = opium_addiction }
		}	
	}
}


# Generate Lists for scripted_sguis
nda_effect_generate_global_cult_list = {

	clear_global_variable_list = nda_global_cult_list

	ordered_country = {
		limit = {
			is_cultist_country = yes
			NOT = { # Hastur does not surrender to a giant cat with squid on its head
				has_variable = nda_king_in_yellow_rules						
			}
		}
		order_by = { 
			subtract = global_country_ranking 
			
			if = {
				limit = { is_full_cult = yes } 
				add = 20
			}
		}
		#min = 1	
		max = 20			
        
		check_range_bounds = no

		add_to_global_variable_list = {
			name = nda_global_cult_list
			target = THIS
		}
	}
}

nda_effect_generate_global_cult_count = {

    set_global_variable = {
        name = nda_topbar_cult_country_count
        value = {
            value = 0
            every_country = {
				limit = {
					is_cultist_country = yes
					NOT = {
						has_variable = nda_king_in_yellow_rules						
					}
				}
                add = 1
            }
        }
    }
}

nda_effect_generate_global_cult_spread_list = {

	clear_global_variable_list = nda_global_cult_spread_list

	ordered_country = {
		limit = {
			is_cultist_country = no
			has_journal_entry = je_nda_cult_spreads
		}
		order_by = { 
			subtract = global_country_ranking 
			
		}
		min = 1	
		max = 20			

		add_to_global_variable_list = {
			name = nda_global_cult_spread_list
			target = THIS
		}
	}
}

nda_effect_generate_global_cult_spread_count = {

    set_global_variable = {
        name = nda_topbar_cult_spread_country_count
        value = {
            value = 0
            every_country = {
				limit = {
					is_cultist_country = no
					has_journal_entry = je_nda_cult_spreads
				}
                add = 1
            }
        }
    }
}

nda_effect_generate_global_coven_leader_list = {
	# just top 5 to use for Conclave ?
	every_country = {
		limit = {
			is_cultist_country = yes
			any_scope_character = {
				has_variable = nda_coven_leader
			}
		}

		random_scope_character = {
			limit = { has_variable = nda_coven_leader }
			add_to_global_variable_list = {
				name = nda_global_cult_list
				target = THIS
			}			
		}
		
	}
}
