nda_effect_home_for_surama = {

	if = {
		limit = { NOT = { any_character = { has_template = nda_surama_character_template } } }

		# find suitable country
		ordered_country = {		

			limit = {
				NOT = { is_country_type = decentralized }
				is_revolutionary = no	
				is_adjacent_to_country = ROOT	
				coa_def_african_trigger = 	yes
				NOT = { has_law_or_variant = law_type:law_isolationism }
			}

			order_by = { 
				subtract = global_country_ranking 
				if = { limit = { has_law_or_variant = law_type:law_isolationism }
					subtract = 10
				}
			}
			max = 1	

			# make room to add character
			if= {
				limit = { empty_agitator_slots = 0 }
				
				random_scope_character = {
					limit = { 
						has_role = agitator 
						# is not existing Altantean
					}
					exile_character_with_martyrdom = yes
				}
			}

			# add surama as agitator to country
			if = {
				limit = {
					NOT = {
						any_scope_character = {
							has_template = nda_surama_character_template
						}
					}
				}

				# # add atlantean acceptance else can be kicked straight away. Not sure this does anything.	
				# capital = {
				# 	add_acceptance = {
				# 		culture = cu:nda_cul_atlantean
				# 		value = 70
				# 	}					
				# }		

				create_character = {
					template = nda_surama_character_template
					interest_group = ig_landowners

					on_created = {					
						set_variable = memento_mori_immunity
						set_character_immortal = yes
						#save_scope_as = scope_sumara

					}
				}

				random_scope_character = {
					limit = { has_template = nda_surama_character_template }

					add_character_role = agitator
				}
				debug_log = "Sumara set as Agitator in [THIS.GetCountry.GetNameNoFormatting]"
			}

			add_journal_entry = {
				type = je_nda_atlantean_surama
			}

			if = {
				limit = { has_global_variable = nda_surama_revive }
				trigger_event = { id = nda_atlantean.11 days = 10 } # They're back! 	
			}
			
		}
	}
}

# Count of Cultures that have 'Unity of Atlantis' modifier
nda_effect_atlantean_unity_list = 
{
	#set_variable = { name = nda_unity_cultures_count value = 0 }
	clear_global_variable_list = nda_unity_cultures_list

	every_country = {
		every_scope_culture = {
			limit = {
				has_discrimination_trait = tradition_nda_grand_union
			}

			add_to_global_variable_list = {
				name = nda_unity_cultures_list
				target = this
			}
		}
	}
}

# Count of Movements that are being boosted by Atlanteans
nda_effect_atlantean_movements_list = 
{
	#set_variable = { name = nda_unity_cultures_count value = 0 }
	clear_global_variable_list = nda_atlantean_movements_list

	every_country = {
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_cultural_minority	
				#culture = { has_discrimination_trait_group =  heritage_group_african }
				has_modifier = nda_covert_support_modifier 
			}

			add_to_global_variable_list = {
				name = nda_atlantean_movements_list
				target = PREV # try adding country, not movement, to list
			}
		}
	}

	
}

nda_effect_atlantean_influence_counts = {

	set_global_variable = { name = nda_atlantean_unity_count value = 0 }
	set_global_variable = { name = nda_atlantean_movements_count value = 0 }
	
	nda_effect_atlantean_unity_list = yes
	nda_effect_atlantean_movements_list = yes

	set_global_variable = {
		name = nda_atlantean_unity_count
		value = {
			value = 0
			every_in_global_list = {
				variable = nda_unity_cultures_list
				add = 1
			}			
		}
	}

	set_global_variable = {
		name = nda_atlantean_movements_count
		value = {
			value = 0
			every_in_global_list = {
				variable = nda_atlantean_movements_list
				add = 1
			}			
		}
	}
}
