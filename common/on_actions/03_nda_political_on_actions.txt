# Root = political movement
on_political_movement_formed = {
	on_actions = {
		nda_noncultist = yes
	}
}

# If Cthulhu is back give malus to nationalist movements in Cult countries
nda_noncultist = {
	effect = {
		if = {
			limit = { 
				owner = { is_cultist_country = yes }	
				OR = {
					is_political_movement_type = movement_cultural_minority
					movement_is_nationalist = yes					
				}			
				culture ?= {
					NOT = { has_discrimination_trait_group = heritage_group_nda_cultist } # must be non cultist
				}
			}

			if = {
				limit = { nda_coa_def_cthulhu_trigger = yes }
				# modifier to supress movement
				add_modifier = nda_reign_of_cthulhu_modifier				
			}
			else_if = {
				limit = { 
					has_global_variable = nda_ritual_cthulhu_leaves 
				}
				# Cthulhu has gone, time for action
				add_modifier = nda_winter_of_the_cults_modifier
			}
		}
	}
}
