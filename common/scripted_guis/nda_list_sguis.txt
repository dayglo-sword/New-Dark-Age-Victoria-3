nda_global_lodge_list_sgui = {
    scope = country

    effect = {
        # custom_description_no_bullet = { 
        #     text = "nda_mason_lodge_list_title_tt"  
        #     object = THIS
        # }

        if = {
            limit = {
                has_global_variable_list = nda_global_lodge_list
            }
            every_in_global_list = {
                variable = nda_global_lodge_list
                save_temporary_scope_as = nda_lodge_loc
                custom_tooltip = "nda_mason_lodge_list_tt" # for each print a line
            }
        }
        else = {
            custom_tooltip = nda_mason_lodge_list_empty_tt # no lodges
        }
    }
}
