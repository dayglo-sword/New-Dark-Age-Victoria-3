	}

		NOT = { age < define:NCharacters|ADULT_AGE }

		exists = owner

			has_variable = nda_cultists_in_open
			NOT = {has_variable = nda_yig_in_open}
		}			
	}
	
	weight = 1
	
	replace = {																
	}
	
	value = 1
}
